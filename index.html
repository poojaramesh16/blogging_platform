/** * FULL SINGLE-FILE BLOGGING PLATFORM (Node.js + Express) * Features: * - Create, View, and Delete posts * - Persistent JSON storage (posts.json) * - All HTML, CSS, JS inline in one file * * Run: * npm install express body-parser fs-extra * node app.js * Then open: http://localhost:3000 */ const express = require("express"); const bodyParser = require("body-parser"); const fs = require("fs-extra"); const path = require("path"); const app = express(); const PORT = 3000; const DATA_FILE = path.join(__dirname, "posts.json"); // ensure file exists if (!fs.existsSync(DATA_FILE)) fs.writeJSONSync(DATA_FILE, []); // middleware app.use(bodyParser.urlencoded({ extended: true })); // load posts function getPosts() { return fs.readJSONSync(DATA_FILE); } // save posts function savePosts(posts) { fs.writeJSONSync(DATA_FILE, posts, { spaces: 2 }); } // serve main page app.get("/", (req, res) => { const posts = getPosts(); res.send(` Node.js Single-File Blog

üìù My Node.js Blog

Create a New Post

Posts

${posts.length === 0 ? "

No posts yet. Write one above!

" : ""} ${posts.map(post => `

${post.title}

${post.content}

üìÖ ${post.date}
`).join("")}
Made with ‚ù§Ô∏è using Node.js (Single-file version)
`); }); // Add new post app.post("/add", (req, res) => { const { title, content } = req.body; if (!title || !content) return res.status(400).send("Missing fields"); const posts = getPosts(); posts.unshift({ id: Date.now().toString(), title, content, date: new Date().toLocaleString(), }); savePosts(posts); res.redirect("/"); }); // Delete post app.post("/delete/:id", (req, res) => { const id = req.params.id; let posts = getPosts(); posts = posts.filter(p => p.id !== id); savePosts(posts); res.redirect("/"); }); app.listen(PORT, () => { console.log(`üöÄ Blogging platform running at http://localhost:${PORT}`); });
